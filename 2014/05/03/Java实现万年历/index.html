Java实现万年历

发表于 2014-05-03   |  

类似于Linux 下的cal -y的效果。直接贴代码。

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.GregorianCalendar;

//输入年份，如2014，打印出该年<span class=’wp_keywordlink_affiliate’><a href=”http://www.maitaidan.com/tag/%e4%b8%87%e5%b9%b4%e5%8e%86“ title=”查看万年历中的全部文章” target=”_blank”>万年历</a></span>，以及标识出当天日期。类似于linux下的cal -y结果。
public class Test8 {
//全局常量 今天的日子~
public static final int toYear = new GregorianCalendar().get(GregorianCalendar.YEAR);
public static final int toMonth = new GregorianCalendar().get(GregorianCalendar.MONTH) + 1;
public static final int today = new GregorianCalendar().get(GregorianCalendar.DAY_OF_MONTH);

public static void main(String[] args) {
    System.out.println("请输入年份");
    BufferedReader bufr = new BufferedReader(new InputStreamReader(
            System.in));
    int year = 0;
    try {
        year = Integer.parseInt(bufr.readLine());
    } catch (Exception e) {
        e.printStackTrace();
    }
    int weekday = firstDay(year);
    System.out.println(weekday);

    // 输出年份
    System.out.println("==========" + year + "年" + "==========");
    // 输出各个月份
    weekday = printMonth(year, weekday);
}

//打印各个月份
private static int printMonth(int year, int weekday) {
    for (int DayInMonth = 1; DayInMonth &lt;= 12; DayInMonth++) {
        printInfo(DayInMonth);
        //打印每个月前面空着的地方
        for (int j = 0; j &lt; weekday; j++) {
            System.out.print("\t\t");
        }
        for (int day = 1; day &lt;= DayOfMonth(year, DayInMonth); day++) {
            //判断是否是今天
            if (year == toYear &amp;&amp; DayInMonth == toMonth &amp;&amp; day == today) {
                System.out.print("★" + day + "★" + "\t");
            } else {
                System.out.print(day + "\t\t");
            }
            //★★★每打印一天后，计算第二天是星期几★★★
            weekday = (weekday + 1) % 7;
            //是星期天就换行
            if (weekday == 0) {
                System.out.println();
            }
        }
        System.out.println();
        System.out.println("--------------------------");
    }
    return weekday;
}

//打印开头的信息
public static void printInfo(int month) {
    System.out.println("【" + month + "月" + "】");
    System.out.println("周日\t周一\t周二\t周三\t周四\t周五\t周六");
}

//判断是否是闰年
public static boolean isLeapYear(int year) {
    GregorianCalendar gc = new GregorianCalendar();
    return gc.isLeapYear(year);
}

// 判断年份的第一天是周几
public static int firstDay(int year) {
    long n = (year - 1900 + 1) * 365;
    for (int i = 1900; i &lt; year; i++) {
        if (isLeapYear(i)) {
            n += 1;
        }
    }
    return (int) n % 7;
}

// 获取每个月的天数
public static int DayOfMonth(int year, int month) {
    switch (month) {
        case 1:
        case 3:
        case 5:
        case 7:
        case 8:
        case 10:
        case 12:
            return 31;
        case 4:
        case 6:
        case 9:
        case 11:
            return 30;
        case 2:
            if (isLeapYear(year))// 判断是否是闰年
                return 29;
            else
                return 28;
        default:
            return 0;
    }
}

}
